  1. Create a new factor variable in the dataset with two levels – “weekday” and “weekend” indicating whether a given date is a weekday or weekend day
df <- df %>% 
  mutate(
    wweek = weekdays(date),
    wweek = case_when(wweek %in% c('sábado', 'domingo') ~ 'weekend',
                      TRUE ~ 'weekday')
